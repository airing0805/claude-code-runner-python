```python
# 引入事件总线
from dataclasses import dataclass
from typing import Callable, Awaitable
import asyncio

@dataclass
class Event:
    type: str
    data: dict
    timestamp: datetime

class EventBus:
    def __init__(self):
        self._handlers: dict[str, list[Callable]] = {}

    def subscribe(self, event_type: str, handler: Callable):
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)

    async def publish(self, event: Event):
        handlers = self._handlers.get(event.type, [])
        await asyncio.gather(*[h(event) for h in handlers])

# 事件类型定义
class EventTypes:
    TASK_STARTED = "task.started"
    TASK_COMPLETED = "task.completed"
    TASK_FAILED = "task.failed"
    SESSION_CREATED = "session.created"
    SESSION_EXPIRED = "session.expired"
```

**收益**:
- 解耦任务执行器和调度器
- 支持实时通知（WebSocket/SSE）

```python
class MessageBuffer:
    def __init__(self, flush_interval: float = 0.1, max_size: int = 10):
        self._buffer: list[StreamMessage] = []
        self._flush_interval = flush_interval
        self._max_size = max_size
        self._last_flush = time.time()

    async def add(self, msg: StreamMessage) -> Optional[list[StreamMessage]]:
        self._buffer.append(msg)
        now = time.time()
        if len(self._buffer) >= self._max_size or now - self._last_flush >= self._flush_interval:
            result = self._buffer
            self._buffer = []
            self._last_flush = now
            return result
        return None
```

```python
# 2. 文件内容缓存
class FileContentCache:
    def __init__(self, ttl: timedelta = timedelta(minutes=5)):
        self._cache: dict[str, tuple[str, float]] = {}
        self._ttl = ttl.total_seconds()

    def get(self, file_path: str) -> Optional[str]:
        if file_path in self._cache:
            content, timestamp = self._cache[file_path]
            if time.time() - timestamp < self._ttl:
                return content
        return None

    def set(self, file_path: str, content: str):
        self._cache[file_path] = (content, time.time())

# 3. 会话文件有效性缓存
_session_valid_cache: dict[str, tuple[bool, float]] = {}

def check_session_file_valid_cached(session_id: str, cwd: str) -> bool:

        valid, timestamp = _session_valid_cache[cache_key]
        if time.time() - timestamp < 60:  # 1 分钟缓存
```

```python
from pydantic import BaseModel, Field, field_validator
import re

class TaskRequest(BaseModel):
    prompt: str = Field(..., min_length=1, max_length=50000)
    working_dir: Optional[str] = Field(None, max_length=500)
    tools: Optional[list[str]] = Field(None, max_items=20)
    timeout: Optional[int] = Field(None, ge=10, le=3600)

    @field_validator('prompt')
    @classmethod
    def validate_prompt(cls, v: str) -> str:
        # 检查危险命令模式
        dangerous_patterns = [
            r'rm\s+-rf\s+/',
            r'del\s+/[sS]',
            r'format\s+',
            r'mkfs',
        ]
        for pattern in dangerous_patterns:
            if re.search(pattern, v, re.IGNORECASE):
                raise ValueError(f'检测到潜在危险命令')
        return v

    @field_validator('working_dir')
    @classmethod
    def validate_working_dir(cls, v: Optional[str]) -> Optional[str]:
        if v is None:
            return v
        # 防止路径遍历
        if '..' in v or '~' in v:
            raise ValueError('工作目录不能包含 .. 或 ~')
        # 规范化路径
        return str(Path(v).resolve())
```

```javascript
class Store {
    constructor(initialState) {
        this.state = initialState;
        this.listeners = new Set();
    }

    getState() {
        return this.state;
    }

    setState(updater) {
        const newState = typeof updater === 'function'
            ? updater(this.state)
            : { ...this.state, ...updater };
```

```javascript
// 1. 虚拟滚动（已实现 virtualList.js，可增强）
class VirtualList {
    constructor(options) {
        this.itemHeight = options.itemHeight || 40;
        this.bufferSize = options.bufferSize || 5;
        this.renderItem = options.renderItem;
        // ...
    }

    render(items, scrollTop) {
        const visibleStart = Math.floor(scrollTop / this.itemHeight);
        const visibleEnd = visibleStart + Math.ceil(this.containerHeight / this.itemHeight);

        const renderStart = Math.max(0, visibleStart - this.bufferSize);
        const renderEnd = Math.min(items.length, visibleEnd + this.bufferSize);

        // 只渲染可见区域 + 缓冲区
        const fragment = document.createDocumentFragment();
        for (let i = renderStart; i < renderEnd; i++) {
            fragment.appendChild(this.renderItem(items[i], i));
        }
        this.container.innerHTML = '';
        this.container.appendChild(fragment);
    }
}

// 2. 防抖/节流优化
const debounce = (fn, delay) => {
    let timer;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => fn(...args), delay);
    };
};

const throttle = (fn, interval) => {
    let lastTime = 0;
    return (...args) => {
        const now = Date.now();
        if (now - lastTime >= interval) {
            fn(...args);
            lastTime = now;
        }
    };
};

// 3. 懒加载
class LazyLoader {
    constructor() {
        this.observer = new IntersectionObserver(entries => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    this._loadElement(entry.target);
                }
            });
        });
    }

    observe(element) {
        this.observer.observe(element);
    }

    _loadElement(element) {
        const src = element.dataset.src;
        if (src) {
            element.src = src;
            this.observer.unobserve(element);
        }
    }
}
```

### 7.3 健康检查端点 (P1)

```python
from pydantic import BaseModel
from enum import Enum

class HealthStatus(str, Enum):
    HEALTHY = "healthy"
    DEGRADED = "degraded"
    UNHEALTHY = "unhealthy"

class HealthCheckResult(BaseModel):
    status: HealthStatus
    version: str
    uptime_seconds: float
    checks: dict[str, dict]

@app.get("/health", response_model=HealthCheckResult)
async def health_check():
    checks = {}

    # Claude API 检查
    try:
        # 简单的 API 可用性检查
        checks["claude_api"] = {"status": "ok"}
    except Exception as e:
        checks["claude_api"] = {"status": "error", "message": str(e)}

    # 调度器检查
    scheduler_status = get_scheduler_status()
    checks["scheduler"] = {
        "status": "ok" if scheduler_status["status"] == "running" else "degraded",
        "queue_count": scheduler_status["queue_count"]
    }

    # 存储检查
    try:
        storage_path = Path("data")
        if storage_path.exists() and os.access(storage_path, os.W_OK):
            checks["storage"] = {"status": "ok"}
        else:
            checks["storage"] = {"status": "error", "message": "Storage not writable"}
    except Exception as e:
        checks["storage"] = {"status": "error", "message": str(e)}

    # 计算整体状态
    if any(c["status"] == "error" for c in checks.values()):
        overall_status = HealthStatus.UNHEALTHY
    elif any(c["status"] == "degraded" for c in checks.values()):
        overall_status = HealthStatus.DEGRADED
    else:
        overall_status = HealthStatus.HEALTHY

    return HealthCheckResult(
        status=overall_status,
        version="0.6.0",
        uptime_seconds=time.time() - START_TIME,
        checks=checks
    )

@app.get("/ready")
async def readiness_check():
    """Kubernetes 就绪检查"""
    scheduler_status = get_scheduler_status()
    if scheduler_status["status"] != "running":
        raise HTTPException(status_code=503, detail="Scheduler not running")
    return {"status": "ready"}

@app.get("/live")
async def liveness_check():
    """Kubernetes 存活检查"""
    return {"status": "alive"}
```
