## 6. 数据迁移实现

### 6.1 迁移检测

```javascript
// 检测是否需要迁移
function needsMigration() {
  const version = loadDataVersion();
  return version !== 'v12';
}

// 执行迁移
async function runMigration() {
  if (!needsMigration()) {
    return false;
  }

  showMigrationProgress(0, '正在准备数据迁移...');

  try {
    // 读取 v11 数据
    const v11Data = loadV11Data();

    showMigrationProgress(30, '正在读取历史数据...');

    // 转换数据
    const v12Data = convertV11ToV12(v11Data);

    showMigrationProgress(60, '正在转换数据格式...');

    // 保存 v12 数据
    saveV12Data(v12Data);

    showMigrationProgress(90, '正在保存迁移数据...');

    // 更新版本标识
    saveDataVersion('v12');

    // 备份原数据
    backupV11Data(v11Data);

    showMigrationProgress(100, '数据迁移完成！');

    setTimeout(() => hideMigrationProgress(), 2000);

    return true;

  } catch (error) {
    console.error('迁移失败:', error);
    showMigrationError(error);
    return false;
  }
}
```

### 6.2 数据转换函数

```javascript
function convertV11ToV12(v11Data) {
  const v12Data = {
    sessions: [],
    currentSessionId: null,
    workspaceHistory: []
  };

  // 转换会话数据
  const sessions = [];
  const sessionMap = new Map();

  for (const [tabId, session] of Object.entries(v11Data.sessions || {})) {
    if (!session || !session.id) continue;

    // 去重处理
    if (sessionMap.has(session.id)) {
      const existing = sessionMap.get(session.id);
      if (session.timestamp > existing.timestamp) {
        sessionMap.set(session.id, session);
      }
    } else {
      sessionMap.set(session.id, session);
    }
  }

  // 转为数组并排序
  v12Data.sessions = Array.from(sessionMap.values())
    .sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));

  // 设置当前会话
  const activeTab = v11Data.activeTab;
  if (activeTab && v11Data.sessions[activeTab]) {
    v12Data.currentSessionId = v11Data.sessions[activeTab].id;
  }

  // 构建工作空间历史
  const workspaceSet = new Set();
  v12Data.sessions.forEach(session => {
    if (session.workingDir) {
      workspaceSet.add(session.workingDir);
    }
  });

  v12Data.workspaceHistory = Array.from(workspaceSet).slice(0, 10);

  return v12Data;
}

function loadV11Data() {
  // 尝试从 localStorage 读取 v11 数据
  const data = localStorage.getItem('sessions_v11');
  if (data) {
    return JSON.parse(data);
  }

  // 兼容其他可能的键名
  const legacyData = localStorage.getItem('sessions');
  if (legacyData) {
    const parsed = JSON.parse(legacyData);
    // 检查是否为 v11 格式
    if (typeof parsed === 'object' && !Array.isArray(parsed)) {
      return parsed;
    }
  }

  return { sessions: {}, activeTab: null };
}
```

### 7.2 前端接口调用

```javascript
// 执行任务
async function executeTask(prompt, options) {
  const response = await fetch('/api/task', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      prompt: prompt,
      working_dir: options.workingDir,
      tools: options.tools,
      permission_mode: options.permissionMode
    })
  });

  if (!response.ok) {
    throw new Error('任务执行失败');
  }

  return await response.json();
}

// 流式执行任务
async function executeTaskStream(prompt, options, onMessage) {
  const response = await fetch(`/api/sessions/${sessionId}/stream`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      prompt: prompt,
      working_dir: options.workingDir,
      tools: options.tools
    })
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    const lines = chunk.split('\n\n');

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = JSON.parse(line.slice(6));
        onMessage(data);
      }
    }
  }
}
```

## 8. 安全考虑

### 8.1 路径验证

```javascript
// 验证工作空间路径
function validateWorkspacePath(path) {
  if (!path || typeof path !== 'string') {
    return { valid: false, error: '路径格式不正确' };
  }

  // 防止路径遍历攻击
  if (path.includes('..') || path.includes('~')) {
    return { valid: false, error: '路径包含非法字符' };
  }

  // 验证路径格式
  if (!path.startsWith('/') && !path.match(/^[A-Za-z]:\\/)) {
    return { valid: false, error: '路径格式不正确' };
  }

  return { valid: true };
}
```

### 8.2 数据清理

```javascript
// 清理无效数据
function cleanSessionData(session) {
  return {
    id: session.id || generateSessionId(),
    workingDir: sanitizePath(session.workingDir),
    messages: session.messages ? session.messages.filter(isValidMessage) : [],
    timestamp: session.timestamp || Date.now(),
    version: 'v12'
  };
}

function isValidMessage(msg) {
  return msg && typeof msg === 'object' && msg.type;
}

function sanitizePath(path) {
  if (!path) return '';
  // 移除查询字符串和片段
  return path.split('?')[0].split('#')[0];
}
```

```javascript
    // 验证历史记录更新
```
