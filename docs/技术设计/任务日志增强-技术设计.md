## 3. 数据模型设计

### 3.1 后端数据模型

#### 3.1.1 LogEntry

**与现有 TaskLog 模型关系**:
- 现有 `TaskLog` 模型使用 `level` (INFO/WARNING/ERROR) 字段
- 新设计使用 `type` (stdout/stderr) 字段区分日志类型
- 两者兼容：`stdout` 对应 INFO，`stderr` 对应 ERROR
- 在存储层统一使用 `type` 字段，在 API 层可兼容转换

```python
@dataclass
class LogEntry:
    """日志条目"""
    id: str                      # 日志唯一标识
    task_id: str                # 任务 ID
    type: str                   # 日志类型: "stdout" | "stderr"
    content: str                # 日志内容
    timestamp: str              # 时间戳 (ISO格式)
    line_number: int            # 行号 (可选)
    
    # 兼容现有 TaskLog 模型的字段
    level: str = "INFO"         # 日志级别（兼容用）
    
    def to_task_log(self) -> "TaskLog":
        """转换为现有 TaskLog 模型"""
        level = "INFO" if self.type == "stdout" else "ERROR"
        return TaskLog(
            id=self.id,
            task_id=self.task_id,
            timestamp=self.timestamp,
            level=level,
            message=self.content,
        )
    
    @classmethod
    def from_task_log(cls, task_log: "TaskLog") -> "LogEntry":
        """从 TaskLog 模型转换"""
        log_type = "stdout" if task_log.level == "INFO" else "stderr"
        return cls(
            id=task_log.id,
            task_id=task_log.task_id,
            type=log_type,
            content=task_log.message,
            timestamp=task_log.timestamp,
            level=task_log.level,
        )
```

     │  SSE 连接 1 │ │ SSE 连接 2 │ │ SSE 连接 N │
     │ (任务 A)    │ │ (任务 B)   │ │ (任务 C)   │
     └─────────────┘ └────────────┘ └────────────┘

### 5.2 实现要点

#### 5.2.1 日志事件发布

```python
# scheduler.py
class Scheduler:
    def __init__(self):
        self.log_events: dict[str, asyncio.Queue] = {}

    def subscribe_logs(self, task_id: str) -> asyncio.Queue:
        """订阅任务日志"""
        if task_id not in self.log_events:
            self.log_events[task_id] = asyncio.Queue()
        return self.log_events[task_id]

    def unsubscribe_logs(self, task_id: str):
        """取消订阅"""
        if task_id in self.log_events:
            del self.log_events[task_id]

    def publish_log(self, task_id: str, log_entry: dict):
        """发布日志事件"""
        if task_id in self.log_events:
            self.log_events[task_id].put_nowait(log_entry)
```

#### 5.2.2 SSE 端点

```python
# scheduler.py
@router.get("/tasks/{task_id}/logs/stream")
async def stream_task_logs(
    task_id: str,
    type: str = Query("all", description="日志类型过滤"),
):
    """SSE 实时日志流"""

    async def event_generator():
        queue = scheduler.subscribe_logs(task_id)

        try:
            while True:
                # 等待日志事件
                log_entry = await queue.get()

                # 根据类型过滤
                if type != "all" and log_entry.get("type") != type:
                    continue

                yield f"data: {json.dumps(log_entry)}\n\n"

                # 检查任务状态
                task = storage.running.get(task_id)
                if not task:
                    # 任务已结束
                    task = storage.history.find_by_id(task_id)
                    if task:
                        if task.status == TaskStatus.COMPLETED:
                            yield "event: complete\n"
                            yield f"data: {json.dumps({'message': '任务已完成'})}\n\n"
                        else:
                            yield "event: error\n"
                            yield f"data: {json.dumps({'message': '任务执行失败', 'error': task.error})}\n\n"
                    yield "event: close\n"
                    yield f"data: {json.dumps({'message': '连接关闭'})}\n\n"
                    break
        finally:
            scheduler.unsubscribe_logs(task_id)

    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",
        },
    )
```

#### 6.2.2 SSE 客户端

```javascript
class SSEClient {
    constructor(url, options = {}) {
        this.url = url;
        this.options = options;
        this.eventSource = null;
        this.retryCount = 0;
        this.maxRetries = 5;
    }

    connect() {
        this.eventSource = new EventSource(this.url);

        // 处理默认消息事件
        this.eventSource.onmessage = (event) => {
            const data = JSON.parse(event.data);
            this.options.onMessage(data);
        };

        // 处理自定义事件: complete
        this.eventSource.addEventListener('complete', (event) => {
            if (event.data) {
                const data = JSON.parse(event.data);
                this.options.onComplete(data);
            }
        });

        // 处理自定义事件: error (任务执行错误)
        this.eventSource.addEventListener('error', (event) => {
            if (event.data) {
                const data = JSON.parse(event.data);
                this.options.onTaskError(data);
            } else {
                this.handleError();
            }
        });

        // 处理自定义事件: close (连接关闭)
        this.eventSource.addEventListener('close', (event) => {
            if (event.data) {
                const data = JSON.parse(event.data);
                this.options.onClose(data);
            }
        });

        // 处理 EventSource 错误
        this.eventSource.onerror = () => {
            this.handleError();
        };
    }

    handleError() {
        if (this.retryCount < this.maxRetries) {
            this.retryCount++;
            const delay = Math.min(3000 * this.retryCount, 15000); // 最大 15 秒
            setTimeout(() => this.connect(), delay);
            this.options.onReconnecting(this.retryCount);
        } else {
            this.options.onMaxRetriesExceeded();
        }
    }

    disconnect() {
        if (this.eventSource) {
            this.eventSource.close();
            this.eventSource = null;
        }
    }
}
```

## 7. 搜索实现设计

### 7.1 搜索策略

#### 7.1.1 关键字搜索

```python
def search_by_keyword(logs: list[dict], keyword: str) -> list[dict]:
    """关键字搜索"""
    results = []
    keyword_lower = keyword.lower()

    for log in logs:
        content = log.get("content", "")
        if keyword_lower in content.lower():
            # 计算匹配位置
            positions = _find_match_positions(content, keyword)
            results.append({
                **log,
                "match_positions": positions
            })

    return results
```

#### 7.1.2 正则搜索

```python
def search_by_regex(logs: list[dict], pattern: str) -> list[dict]:
    """正则表达式搜索"""
    import re

    results = []
    try:
        regex = re.compile(pattern, re.IGNORECASE)
    except re.error:
        return results  # 无效正则，返回空

    for log in logs:
        content = log.get("content", "")
        match = regex.search(content)
        if match:
            results.append({
                **log,
                "match_positions": [[match.start(), match.end()]]
            })

    return results
```

### 7.2 匹配位置计算

```python
def _find_match_positions(content: str, keyword: str) -> list[list[int]]:
    """计算关键字在内容中的位置"""
    positions = []
    keyword_lower = keyword.lower()
    content_lower = content.lower()

    start = 0
    while True:
        pos = content_lower.find(keyword_lower, start)
        if pos == -1:
            break
        positions.append([pos, pos + len(keyword)])
        start = pos + 1

    return positions
```
