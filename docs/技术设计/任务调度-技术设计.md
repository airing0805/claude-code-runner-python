class FileLock:
    """文件锁实现"""
    
    def acquire(self, blocking: bool = True) -> bool:
        """获取文件锁（原子创建）"""
        try:
            fd = os.open(lock_path, os.O_CREAT | os.O_EXCL | os.O_WRONLY)
            os.close(fd)
            return True
        except FileExistsError:
            # 锁已被占用
            return False
    
    def release(self) -> None:
        """释放文件锁"""
        lock_path.unlink()

def atomic_write(filepath: Path, data: dict) -> None:
    """原子写入 JSON 文件"""
    # 1. 写入临时文件
    fd, temp_path = tempfile.mkstemp(dir=filepath.parent, suffix=".tmp")
    with os.fdopen(fd, "w") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    
    # 2. 原子重命名
    os.replace(temp_path, filepath)

## 五、调度器核心设计

### 5.1 调度循环

```python
async def _run_loop(self) -> None:
    """主调度循环"""
    while not self._stop_event.is_set():
        try:
            # 1. 检查定时任务到期
            await self._check_scheduled_tasks()
            
            # 2. 执行队列中的任务
            await self._process_queue()
            
        except Exception as e:
            logger.exception(f"调度循环异常: {e}")
        
        # 等待下一次轮询（默认10秒）
        await asyncio.wait_for(
            self._stop_event.wait(),
            timeout=self.poll_interval,
        )
```

### 5.2 定时任务检查

```python
async def _check_scheduled_tasks(self) -> None:
    """检查定时任务是否到期"""
    scheduled_tasks = self.storage.scheduled.get_enabled()
    
    for scheduled in scheduled_tasks:
        # 检查是否到期（next_run <= 当前时间）
        if is_due(scheduled.next_run):
            # 防止并发重复触发
            latest = self.storage.scheduled.get(scheduled.id)
            if latest and is_due(latest.next_run):
                await self._trigger_scheduled_task(latest)
```

### 5.3 任务执行流程

```python
async def _execute_task(self, task: Task) -> ExecutionResult:
    """执行单个任务"""
    
    # 1. 验证任务
    if not self._validate_task(task):
        return ExecutionResult(success=False, error="验证失败")
    
    # 2. 更新状态为运行中
    task.status = TaskStatus.RUNNING
    task.started_at = datetime.now().isoformat()
    self.storage.running.add(task)
    
    try:
        # 3. 执行（带超时控制）
        result = await asyncio.wait_for(
            self._execute_with_client(task),
            timeout=task.timeout / 1000.0,
        )
        
        if result.success:
            return self._handle_success(task, result)
        else:
            return self._handle_failure(task, result)
            
    except asyncio.TimeoutError:
        return self._handle_timeout(task)
    except Exception as e:
        return self._handle_error(task, e)
```

### 6.3 指数退避延迟

```python
def calculate_retry_delay(retry_count: int) -> float:
    """计算重试延迟（指数退避 + 抖动）"""
    # 指数退避：delay = base * 2^retry_count
    delay = BASE_DELAY * (2 ** retry_count)
    
    # 限制最大延迟
    delay = min(delay, MAX_DELAY)
    
    # 添加随机抖动（±10%）
    jitter = delay * JITTER * random.uniform(-1, 1)
    delay = delay + jitter
    
    return max(0, delay)
```

### 7.4 下次执行时间计算

```python
def calculate_next_run(self, cron: str, from_time: datetime | None) -> datetime | None:
    """计算下次执行时间"""
    expression = self.parse(cron)
    
    # 5位格式从下一分钟开始，6位格式从下一秒开始
    if not expression.is_extended:
        current = from_time.replace(second=0, microsecond=0) + timedelta(minutes=1)
    else:
        current = from_time.replace(microsecond=0) + timedelta(seconds=1)
    
    # 迭代查找下一个匹配时间（最多1年）
    for _ in range(366 * 24 * 60):
        if self._matches(expression, current):
            return current
        current += timedelta(minutes=1 if not expression.is_extended else seconds=1)
    
    return None
```

    # 记录错误

    # 判断是否重试

## 九、安全控制设计

### 9.1 工作目录验证

```python
def validate_workspace(workspace: str | None) -> str:
    """验证工作目录安全性"""
    if workspace is None:
        return "."
    
    # 1. 路径规范化
    abs_workspace = Path(workspace).resolve()
    
    # 2. 检查路径遍历攻击
    if ".." in workspace:
        raise ValueError("不允许使用路径遍历")
    
    # 3. 检查系统敏感目录
    forbidden_dirs = ["/etc", "/root", "C:\\Windows", "C:\\System32"]
    for forbidden in forbidden_dirs:
        if str(abs_workspace).startswith(forbidden):
            raise ValueError("禁止访问系统目录")
    
    # 4. 白名单检查
    base_dir = Path(os.getenv("WORKING_DIR", "")).resolve()
    try:
        abs_workspace.relative_to(base_dir)
    except ValueError:
        raise ValueError("工作目录必须在允许范围内")
    
    return str(abs_workspace)
```

async def stop(self) -> bool:
    """停止调度器（优雅停止）"""
    self._status = SchedulerStatus.STOPPING
    self._stop_event.set()
    
    # 等待当前任务完成（最多5秒）
    if self._task and not self._task.done():
        try:
            await asyncio.wait_for(self._task, timeout=5.0)
        except asyncio.TimeoutError:
            self._task.cancel()
    
    self._status = SchedulerStatus.STOPPED
    return True
