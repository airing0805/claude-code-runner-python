```

│  app/routers/auth.py  │  app/routers/keys.py  │  app/routers/admin.py │
└─────────────────────────────┬───────────────────────────────┘
                              │
┌─────────────────────────────▼───────────────────────────────┐
│                     业务逻辑层                               │
│  app/services/auth_service.py  │  app/services/key_service.py │
```

## 三、认证中间件设计

### 3.1 认证依赖注入

```python
from fastapi import Depends, HTTPException, Request
from fastapi.security import APIKeyHeader, BearerTokenCredentials
from typing import Optional

class AuthDependency:
    """认证依赖注入类"""

    def __init__(
        self,
        api_key_header: Optional[str] = APIKeyHeader(None, name="X-API-Key"),
        bearer_token: Optional[str] = BearerTokenCredentials(None),
    ):
        self.api_key = api_key_header
        self.bearer_token = bearer_token

async def get_current_user(
    auth: AuthDependency = Depends(),
    auth_service: AuthService = Depends(get_auth_service)
) -> User:
    """获取当前用户依赖"""

    # 优先尝试 API Key 认证
    if auth.api_key:
        user = await auth_service.authenticate_api_key(auth.api_key)
        if user:
            return user

    # 其次尝试 Bearer Token 认证
    if auth.bearer_token:
        user = await auth_service.authenticate_token(auth.bearer_token)
        if user:
            return user

    # 认证失败
    raise HTTPException(
        status_code=401,
        detail="无效的认证凭证",
        headers={"WWW-Authenticate": "Bearer"}
    )

async def require_admin(
    user: User = Depends(get_current_user)
) -> User:
    """管理员权限依赖"""
    if not user.is_admin:
        raise HTTPException(
            status_code=403,
            detail="需要管理员权限"
        )
    return user
```

### 3.2 认证流程

```
请求进入
    ↓
检查 X-API-Key
    ↓ [存在]
调用 auth_service.authenticate_api_key(api_key)
    ↓ [成功]
检查用户 is_active
    ↓ [是]
返回 User 对象
    ↓ [否]
返回 403 Forbidden

    ↓ [不存在]
检查 Authorization: Bearer
    ↓ [存在]
调用 auth_service.authenticate_token(token)
    ↓ [成功]
检查用户 is_active
    ↓ [是]
返回 User 对象
    ↓ [否]
返回 403 Forbidden

    ↓ [无效或不存在]
返回 401 Unauthorized
```

```python
@router.post("/register", status_code=201, response_model=UserResponse)
async def register(request: RegisterRequest) -> User:
    """
    用户注册

    验证:
    - username: 邮箱格式验证
    - password: 长度 ≥8，含字母+数字
    - name: 非空

    错误:
    - 400: 用户名已存在
    - 422: 参数验证失败
    """
    # 1. 验证用户名格式
    validate_email(request.username)

    # 2. 验证密码复杂度
    validate_password(request.password)

    # 3. 检查用户名是否已存在
    existing = await user_repo.get_by_username(request.username)
    if existing:
        raise HTTPException(400, "用户名已存在")

    # 4. 密码哈希
    password_hash = hash_password(request.password)

    # 5. 创建用户
    user = User(
        username=request.username,
        password_hash=password_hash,
        name=request.name
    )
    await user_repo.create(user)

    # 6. 返回响应（不包含密码）
    return UserResponse(**user.to_dict())
```

#### 4.1.2 用户登录

```python
@router.post("/login", response_model=LoginResponse)
async def login(request: LoginRequest, request_obj: Request) -> LoginResult:
    """
    用户登录

    验证:
    - username: 存在且正确
    - password: 匹配
    - IP 限流

    错误:
    - 401: 用户名或密码错误
    - 429: 登录尝试过于频繁
    """
    # 1. IP 限流检查
    client_ip = request_obj.client.host
    await rate_limiter.check(client_ip)

    # 2. 获取用户
    user = await user_repo.get_by_username(request.username)
    if not user or not verify_password(request.password, user.password_hash):
        await rate_limiter.record(client_ip)
        raise HTTPException(401, "用户名或密码错误")

    # 3. 检查用户状态
    if not user.is_active:
        raise HTTPException(403, "账号已被禁用")

    # 4. 更新最后登录时间
    user.last_login = datetime.now()
    await user_repo.update(user)

    # 5. 生成 Token
    access_token = create_access_token(user.user_id)
    refresh_token = create_refresh_token(user.user_id)

    return LoginResult(
        access_token=access_token,
        refresh_token=refresh_token
    )
```

```python
@router.post("", status_code=201, response_model=APIKeyCreateResponse)
async def create_key(
    request: CreateKeyRequest,
    current_user: User = Depends(get_current_user)
) -> APIKeyCreateResult:
    """
    创建 API 密钥

    注意: 完整密钥只在响应中返回一次，之后无法找回
    """
    # 1. 生成密钥
    key = generate_api_key()  # sk-ccr-{32位随机字符}

    # 3. 创建密钥记录
    api_key = APIKey(
        user_id=current_user.user_id,
        key_hash=key_hash,
        prefix=prefix,
        name=request.name,
        expires_at=expires_at
    )
    await key_repo.create(api_key)

    # 4. 返回完整密钥（仅此一次）
    return APIKeyCreateResult(
        key_id=api_key.key_id,
        name=api_key.name,
        key=key,  # ⚠️ 仅在此处返回
        created_at=api_key.created_at,
        expires_at=api_key.expires_at,
        is_active=api_key.is_active
    )
```

#### 4.2.3 撤销密钥

```python
@router.delete("/{key_id}", status_code=204)
async def revoke_key(
    key_id: str,
    current_user: User = Depends(get_current_user)
) -> None:
    """撤销 API 密钥

    撤销流程:
    1. 验证密钥归属（确保用户只能撤销自己的密钥）
    2. 将密钥状态标记为 inactive
    3. 将密钥 ID 加入黑名单（确保已撤销密钥立即失效）
    """
    # 1. 获取密钥
    key = await key_repo.get_by_id(key_id)
    if not key or key.user_id != current_user.user_id:
        raise HTTPException(404, "密钥不存在")

    # 2. 撤销密钥 - 标记为 inactive
    key.is_active = False
    await key_repo.update(key)

    # 3. 将密钥 ID 加入黑名单
    # 使用 key_id 作为标识符，过期时间取密钥本身的过期时间或当前时间+1小时
    expires_at = key.expires_at or (datetime.now() + timedelta(hours=1))
    await token_blacklist.add(key.key_id, key.user_id, expires_at)

    return None
```

## 五、安全控制设计

### 5.1 密码安全

```python
import re
from passlib.context import CryptContext

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def hash_password(password: str) -> str:
    """密码哈希"""
    return pwd_context.hash(password)

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """验证密码"""
    return pwd_context.verify(plain_password, hashed_password)

def validate_password(password: str) -> None:
    """密码验证"""
    if len(password) < 8:
        raise ValueError("密码长度至少 8 位")
    if not re.search(r"[a-zA-Z]", password):
        raise ValueError("密码必须包含字母")
    if not re.search(r"[0-9]", password):
        raise ValueError("密码必须包含数字")
```

### 5.2 API 密钥安全

```python
import secrets
import hashlib

def generate_api_key() -> str:
    """生成 API 密钥
    格式: sk-ccr-{32位随机字符串}
    总长度: 39 字符 (7 + 32)
    """
    random_part = secrets.token_urlsafe(24)  # 生成约 32 个 URL 安全字符
    return f"sk-ccr-{random_part}"

def hash_api_key(key: str) -> str:
    """API 密钥哈希"""
    return hashlib.sha256(key.encode()).hexdigest()

def get_key_prefix(key: str) -> str:
    """获取密钥前缀"""
    return key[:12]  # sk-ccr-xxx...
```

### 5.3 JWT Token 安全

```python
from jose import jwt, JWTError
from datetime import datetime, timedelta

SECRET_KEY = os.getenv("JWT_SECRET_KEY", "your-secret-key")
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE = 3600  # 1 小时
REFRESH_TOKEN_EXPIRE = 604800  # 7 天

def create_access_token(user_id: str) -> str:
    """创建 Access Token"""
    now = datetime.utcnow()
    payload = {
        "sub": user_id,
        "type": "access",
        "iat": now,
        "exp": now + timedelta(seconds=ACCESS_TOKEN_EXPIRE)
    }
    return jwt.encode(payload, SECRET_KEY, algorithm=ALGORITHM)

def create_refresh_token(user_id: str) -> str:
    """创建 Refresh Token"""
    now = datetime.utcnow()
    jti = secrets.token_urlsafe(16)
    payload = {
        "sub": user_id,
        "type": "refresh",
        "jti": jti,
        "iat": now,
        "exp": now + timedelta(seconds=REFRESH_TOKEN_EXPIRE)
    }
    return jwt.encode(payload, SECRET_KEY, algorithm=ALGORITHM)

def verify_token(token: str, expected_type: str = "access") -> Optional[dict]:
    """验证 Token
    注意: 需要在调用处注入 token_blacklist 服务进行黑名单检查
    """
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        if payload.get("type") != expected_type:
            return None
        return payload
    except JWTError:
        return None

# 完整的 Token 验证（包括黑名单检查）应在 AuthService 中调用
async def verify_token_with_blacklist(
    token: str,
    expected_type: str,
    token_blacklist: TokenBlacklistStorage
) -> Optional[dict]:
    """验证 Token 并检查黑名单"""
    payload = verify_token(token, expected_type)
    if not payload:
        return None

    # 检查 Token 是否在黑名单中
    jti = payload.get("jti")
    if jti and await token_blacklist.is_blacklisted(jti):
        return None

    return payload
```

### 5.4 限流控制

```python
from collections import defaultdict
from datetime import datetime, timedelta
import asyncio

class RateLimiter:
    """登录限流器"""

    def __init__(self, max_attempts: int = 10, window_seconds: int = 300):
        self.max_attempts = max_attempts
        self.window_seconds = window_seconds
        self._records: dict[str, RateLimitRecord] = defaultdict(
            lambda: RateLimitRecord(ip_address="", reset_at=datetime.now())
        )
        self._lock = asyncio.Lock()

    async def check(self, ip: str) -> None:
        """检查是否允许登录"""
        async with self._lock:
            record = self._records[ip]
            now = datetime.now()

            # 检查是否在限制期内
            if now < record.reset_at:
                if record.attempt_count >= self.max_attempts:
                    retry_after = int((record.reset_at - now).total_seconds())
                    raise HTTPException(
                        status_code=429,
                        detail="登录尝试过于频繁，请 5 分钟后再试",
                        headers={"Retry-After": str(retry_after)}
                    )
            else:
                # 重置记录
                record.attempt_count = 0
                record.first_attempt = now
                record.reset_at = now + timedelta(seconds=self.window_seconds)

    async def record(self, ip: str) -> None:
        """记录失败登录"""
        async with self._lock:
            record = self._records[ip]
            record.attempt_count += 1
```

```python
        async with aiofiles.open(self.users_file, "a") as f:
            await f.write(json.dumps(user.__dict__) + "\n")

    async def get_by_username(self, username: str) -> Optional[User]:
        """按用户名查询"""
        async with aiofiles.open(self.users_file, "r") as f:
            async for line in f:
                user_data = json.loads(line)
                if user_data["username"] == username:
                    return User(**user_data)
        return None

    async def get_by_id(self, user_id: str) -> Optional[User]:
        """按 ID 查询"""
        async with aiofiles.open(self.users_file, "r") as f:
            async for line in f:
                user_data = json.loads(line)
                if user_data["user_id"] == user_id:
                    return User(**user_data)
        return None

    async def update(self, user: User) -> None:
        """更新用户"""
        # 读取所有用户
        users = []
        async with aiofiles.open(self.users_file, "r") as f:
            async for line in f:
                user_data = json.loads(line)
                if user_data["user_id"] == user.user_id:
                    users.append(user.__dict__)
                else:
                    users.append(user_data)

        # 写回文件
        async with aiofiles.open(self.users_file, "w") as f:
            for user_data in users:
                await f.write(json.dumps(user_data) + "\n")

    async def list_paginated(self, page: int = 1, limit: int = 20) -> tuple[list[User], int]:
        """分页获取用户列表
        返回: (用户列表, 总数)
        """
        users = []
        total = 0

        async with aiofiles.open(self.users_file, "r") as f:
            async for line in f:
                total += 1
                # 跳过不需要的页
                skip = (page - 1) * limit
                if skip > 0:
                    skip -= 1
                    continue
                # 达到限制数量后停止
                if len(users) >= limit:
                    break

                user_data = json.loads(line)
                users.append(User(**user_data))

        return users, total
```

```python
        """检查是否在黑名单中"""
        async with aiofiles.open(self.blacklist_file, "r") as f:
            async for line in f:
                entry = json.loads(line)
                if entry["jti"] == jti:
                    return True
        return False

    async def cleanup(self) -> None:
        """清理过期条目"""
        now = datetime.now()
        valid_entries = []

        async with aiofiles.open(self.blacklist_file, "r") as f:
            async for line in f:
                entry = json.loads(line)
                expires_at = datetime.fromisoformat(entry["expires_at"])
                if expires_at > now:
                    valid_entries.append(entry)

        async with aiofiles.open(self.blacklist_file, "w") as f:
            for entry in valid_entries:
                await f.write(json.dumps(entry) + "\n")
```
